import cv2
import os

IMG1_PATH = "/home/wintermute/image/image1.jpeg"
IMG2_PATH = "/home/wintermute/image/image2.jpeg"

if not os.path.exists(IMG1_PATH) or not os.path.exists(IMG2_PATH):
    print("Error: image file not found.")
    exit()

img1 = cv2.imread(IMG1_PATH)
img2 = cv2.imread(IMG2_PATH)

if img1 is None or img2 is None:
    print("Error: failed to read one of the images.")
    exit()

try:
    feature = cv2.SIFT_create()
    print("Using SIFT")
except Exception:
    print("SIFT not available, using ORB")
    feature = cv2.ORB_create(nfeatures=2000)

kp1, des1 = feature.detectAndCompute(img1, None)
kp2, des2 = feature.detectAndCompute(img2, None)

if des1 is None or des2 is None:
    print("Error: no descriptors found.")
    exit()

if isinstance(feature, cv2.ORB):
    bf = cv2.BFMatcher(cv2.NORM_HAMMING, crossCheck=False)
else:
    bf = cv2.BFMatcher(cv2.NORM_L2, crossCheck=False)

matches = bf.knnMatch(des1, des2, k=2)

good = []
for m, n in matches:
    if m.distance < 0.6 * n.distance:
        good.append(m)

print("Good matches:", len(good))

img3 = cv2.drawMatchesKnn(img1, kp1, img2, kp2, [good], None,
    flags=cv2.DRAW_MATCHES_FLAGS_NOT_DRAW_SINGLE_POINTS)

h, w, c = img3.shape
ratio = float(h) / float(w)
img3 = cv2.resize(img3, (1024, int(1024 * ratio)))

cv2.imshow("matches", img3)
cv2.waitKey(0)
cv2.destroyAllWindows()
