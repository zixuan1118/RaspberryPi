import cv2

frame = cv2.imread('/home/wintermute/image6.jpeg')
if frame is None:
    raise FileNotFoundError("Could not find '/home/wintermute/image6.jpeg'. Please make sure the file exists.")

gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)

edged = cv2.Canny(gray, 50, 150)
edged = cv2.dilate(edged, None, iterations=1)

contours, hierarchy = cv2.findContours(
    edged,
    cv2.RETR_EXTERNAL,
    cv2.CHAIN_APPROX_SIMPLE
)

if len(contours) == 0:
    raise ValueError("No contours detected!")

cnt = max(contours, key=cv2.contourArea)

cnt = cv2.approxPolyDP(cnt, 30, True)

hull = cv2.convexHull(cnt, returnPoints=False)
defects = cv2.convexityDefects(cnt, hull)

if defects is None:
    print("?? No convexity defects found (the shape might be fully convex).")
    defects = []

print("Number of convex points:", len(hull))
print("Number of concave defects:", len(defects))

if len(defects) > 0:
    for i in range(defects.shape[0]):
        s, e, f, d = defects[i, 0]
        start = tuple(cnt[s][0])
        end = tuple(cnt[e][0])
        far = tuple(cnt[f][0])

        cv2.line(frame, start, end, (0, 255, 0), 2)    
        cv2.circle(frame, far, 5, (0, 0, 255), -1)      

cv2.imwrite("output.png", frame)
print("Output image saved as output.png")
